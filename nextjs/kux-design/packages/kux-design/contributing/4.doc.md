### `Callout` 突出提示, 用于着重突出一些信息
```tsx
export interface ICalloutProps {
   /** 
   * callout 类型
   * * note: 一般用于提示信息, 默认值
   * * tip: 一般用于小技巧提示
   * * warning: 一般用于警告信息
   * * danger: 一般用于错误/危险信息
   */
  type?: 'note' | 'tip' | 'warning' | 'danger';
  /**
   * 标题
   * 如果不传则使用默认标题
   */
  title?: string;
  /** 
   * 内容
   */
  children: React.ReactNode;
}

// 使用示例:
<Callout type="tip">
   there is some useful tips...
</Callout>
```
